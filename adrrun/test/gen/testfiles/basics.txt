$START("ifelse")
int loop() {
    int a = 0;
    for(int i in [1, 2, 3, 4]) {
        if( i >= 4 ) {
            return i + a;
        } else if ( i == 1 ) {
            a = a + 2;
        } else if ( i == 2 ) {
            a = a - 1;
        } else {
            a = a - 1;
        }
    }
    return 0;
}
int main() {
    return loop();
}
