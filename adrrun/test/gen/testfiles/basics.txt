$START("ifelse")

    num a = 0;
    for(num i in [1, 2, 3, 4]) {
        if( i >= 4 ) {
            return i + a;
        } else if ( i == 1 ) {
            a = a + 2;
        } else if ( i == 2 ) {
            a = a - 1;
        } else {
            a = a - 1;
        }
    }
    return 0;
}
num main() {
    return loop();
}
